### 斐波那契数列

##### 递归

```js
function fb(n) {
  if (n <= 1) return n;
  return fb(n - 1) + f(n - 2);
}
```

##### 循环

```js
function fb(n) {
  if (n <= 1) return n;
  let pre = 0,
    current = 1;
  for (let i = 2; i <= n; i++) {
    let temp = current;
    current = prev + current;
    prev = temp;
  }
  return current;
}
// 时间复杂度: O(n)
// 空间复杂度: O(1)
```

##### 动态规划

```js
function fb(n) {
  if (n <= 1) return n;
  let dp = [0, 1];
  for (let i = 2; i <= n; i++) {
    dp[i] = dp[i - 1] + dp[i - 2];
  }
  return dp[n];
}
// 时间复杂度: O(n)
// 空间复杂度: O(n)
```
