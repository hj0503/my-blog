### 防抖

```js
function debounce(fn, delay) {
  let timer = null;
  return function (...args) {
    if (timer) {
      clearTimeout(timer);
    }
    timer = setTimeout(() => {
      fn.call(this, ...args);
    }, delay);
  };
}
```

### 节流

```js
function throttle(fn, delay) {
  let bef = 0;
  return function (...args) {
    const now = new Date().getTime();

    if (now - bef > delay) {
      fn.call(this, ...args);
      bef = now;
    }
  };
}
```
